### rufus-scheduler
---
https://github.com/jmettraux/rufus-scheduler

```ruby
#quickstart.rb

require 'rufus-scheduler'
scheduler = Rufus::Scheduler.new
scheduler.in '3s' do
  puts 'Hello'
end
scheduler.join

require 'rugus-scheduler'
scheduler = Rufus::Scheduler.new
scheduler.in '10d' do
end
scheduler.every '3h' do
end
scheduler.every '' do
end
scheduler.cron '' do
end
```
